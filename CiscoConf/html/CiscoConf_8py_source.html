422  def logsplit_dirdir(self, indir=None, otdir=None):
423  """ cisco-logを indir/DIRS/*.txt から otdir/DIRS/*.txt.ext.txtに分解
424 
425  :param indir :str, 入力ディレクトリ
426  :param otdir :str, 出力ディレクトリ。otdir/DIRSはあらかじめ作成する必要がある。
427  """
428  for indir1 in os.listdir(indir):
429  inpath = indir + '/' + indir1
430  otpath = otdir + '/' + indir1
431  for infname in os.listdir(inpath):
432  m = re.match(r'^(.*)(_\d+_\d+).txt$', infname)
433  if m is None:
434  self.dbgprint('logsplit_dirdir', '', 'Not Target file:' + inpath + ' ' + infname)
435  else:
436  (base, els) = m.groups()
437  logf = inpath + '/' + infname
438  self.logsplit(logf, otdir=otpath, otfbase=base)
439  pass

471  def search_ip_int2(self, ip, confname):
472  """ ipをint2から検索し、リストで返却
473 
474  :param ip :str, ip-address
475  :param confname :str, CiscoConfig file
476  :return list [confname, [intf, linkst, protocolst, address, desc, ipaddress]]
477  """
478  confobj = self.read_conf(confname) # .read()
479  parsed = self.parse(confobj, 'int') # parse with template
480  rtn = []
481  for _int in parsed:
482  intf = _int[eInt.interface]
483  linkst = _int[eInt.link_status]
484  protocolst = _int[eInt.protocol_status]
485  address = _int[eInt.address]
486  desc = _int[eInt.description]
487  net = _int[eInt.ip_address]
488  if net != '':
489  ip_address = ipaddress.IPv4Address(unicode(ip, 'utf-8'))
490  network = ipaddress.IPv4Network(unicode(net, 'utf-8'), strict=False)
491  _int2 = [intf, linkst, protocolst, address, desc, net]
492  if ip_address in network:
493  rtn.append([confname, _int2])
494  else:
495  return rtn

728  def search_ip_eigrptop(self, ip, confname):
729  """ ipをeigrptopから検索し、リストで返却
730 
731  :param ip :str, ip-address
732  :param confname :str, CiscoConfig file

734  """

742  network = ipaddress.IPv4Network(unicode(net, 'utf-8'))

844  def nextip_eigrptop(self, confname):
845  """ confnameをeigrptopで解析し、自分->次ルータを検索しdigraph作成用オブジェクトを設定する
846 
847  一つのconfname(=router)を処理する
848  for node : self.asno_router[(asno, router)] = self.confname2base(confname)を設定
849  for edge : NONAME-DIC[(asno,router,_nextip)] = cnt を返却
850  router is from, _netxip is to
851 
852  :param confname :str, a confname
853  :return :dic, key=(asno, router, nextip), val=count in confname
854  """
855  confobj = self.read_conf(confname) # .read()
856  parsed = self.parse(confobj, 'eigrptop') # parse with template
857  rtn = {}
858  _asno = ''
859  _router = ''
860  for eigrp in parsed:
861  [asno, router, code, route, route_mask, successors, fd, tag, serno, adv, outif] = eigrp
862  if _asno != asno or _router != router:
863  key = '%s_%s' % (asno, router)
864  if key in self.asno_router:
865  fname = os.path.basename(confname)
866  _msg = 'Key dup\n%s->%s in %s' % \
867  (key, self.asno_router[key], fname)
868  self.errprint('nextip_eigrptop', 'CiscoConf', _msg)
869  pass
870  self.asno_router[key] = self.confname2base(os.path.basename(confname))
871  _asno = asno
872  _router = router
873  pass
874  for via_nextip in adv:
875  nextip = via_nextip.split() # via_nextip is '\S+' or 'ip-addr (addis/metric)'
876  _nextip = nextip[0]
877  if re.match(r'\d+\.\d+\.\d+\.\d+', _nextip) is not None:
878  if not (asno, router, _nextip) in rtn:
879  rtn[(asno, router, _nextip)] = 1
880  else:
881  rtn[(asno, router, _nextip)] += 1
882  pass
883  pass
884  return rtn

887  """ confnamesを順にeigrptopで解析し、self.router_nextip[]を設定する。
888
