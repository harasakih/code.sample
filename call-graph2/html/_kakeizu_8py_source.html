368  def sison_maketree(self, level, parent, kiten, stop=None):
369  """ 自分の子孫を深さ優先で探索しリスト"kktree"を作成する
370  [level,親,自分,err-flg]を self.kktree に追加し、
371  自分の子供を再帰的に [level+1,自分,子供,err-flg] 検索する
372  :param int level: 検索レベル
373  :param str parent: 親から自分が呼ばれた
374  :param str kiten: 自分
375  :param stop: 検索停止条件、正規表現
376  :rtype: int
377  :return: 0 正常 / 1 検索停止あり
378  """
379  if level == 0:
380  self.me = kiten
381  self.kktree = []
382  self.visited.clear()
383  self.visited2.clear()
384 
385  # レベル０で、辞書なしは指定誤り
386  if level == 0 and kiten not in self.sison:
387  self.kktree.append([level, parent, kiten, 'NON'])
388  return 1
389 
390  # 探索対象がヌル
391  if not kiten:
392  self.kktree.append([level, parent, kiten, 'NUL'])
393  return 1
394 
395  # 一定以上の深さは、検索を中止する
396  if level >= self.maxdepth:
397  self.visited[kiten] = 1
398  self.kktree.append([level, parent, kiten, 'MAX'])
399  return 1
400 
401  # 検索済の場合は、中止する
402  # 2017/06/30 フラグ＝1のみ、チェック実施
403  if self.isdup != 0 and kiten in self.visited:
404  self.kktree.append([level, parent, kiten, 'DUP'])
405  return 1
406 
407  # 中止条件がある場合は、中止する
408  if stop is not None and re.match(stop, kiten):
409  self.visited[kiten] = 1
410  self.kktree.append([level, parent, kiten, 'STP'])
411  return 1
412 
413  # 自分の子供を再帰検索する
414  self.visited[kiten] = 1
415  self.kktree.append([level, parent, kiten, ''])
416 
417  level2 = level + 1
418  if kiten in self.sison:
419  for child in self.sison[kiten].split(';'):
420  if child:
421  self.sison_maketree(level2, kiten, child, stop)
